/**
 * Determina el nombre del mes basado en su número (1-12)
 * @param {number} numeroMes - Número de mes (1-12)
 * @returns {string} Nombre del mes en español
 */
function obtenerNombreMes(numeroMes) {
  const nombresMeses = [
    'Enero', 'Febrero', 'Marzo', 'Abril', 'Mayo', 'Junio',
    'Julio', 'Agosto', 'Septiembre', 'Octubre', 'Noviembre', 'Diciembre'
  ];
  
  // Validar rango
  if (numeroMes < 1 || numeroMes > 12) {
    throw new Error(`Número de mes inválido: ${numeroMes}`);
  }
  
  return nombresMeses[numeroMes - 1];
}
 
/**
 * Extrae información de mes y año del nombre del archivo
 * @param {string} nombreArchivo - Nombre del archivo
 * @returns {Object|null} - Objeto con mes y año o null si no se puede extraer
 */
function extraerPeriodoDeNombreArchivo(nombreArchivo) {
  if (!nombreArchivo) return null;
  
  // Buscar patrones como "Dimensionamiento_May_2025.csv"
  const matchNombreMes = nombreArchivo.match(/Dimensionamiento_([A-Za-z]+)_(\d{4})/i);
  if (matchNombreMes) {
    const abrevMes = matchNombreMes[1].toLowerCase();
    const anio = parseInt(matchNombreMes[2]);
    
    // Mapeo de abreviaturas a números de mes
    const mesesAbrev = {
      'ene': 1, 'jan': 1,
      'feb': 2,
      'mar': 3,
      'abr': 4, 'apr': 4,
      'may': 5,
      'jun': 6,
      'jul': 7,
      'ago': 8, 'aug': 8,
      'sep': 9,
      'oct': 10,
      'nov': 11,
      'dic': 12, 'dec': 12
    };
    
    // Buscar coincidencia con inicio de abreviatura
    for (const [abrev, numMes] of Object.entries(mesesAbrev)) {
      if (abrevMes.startsWith(abrev)) {
        return { mes: numMes, anio, nombreMes: obtenerNombreMes(numMes) };
      }
    }
  }
  
  return null;
}
 
/**
 * Normaliza y valida un formato de hora HH:MM
 * @param {string} horaStr - String con la hora en formato HH:MM
 * @returns {string|null} - Hora normalizada o null si es inválida
 */
function normalizarHora(horaStr) {
  if (!horaStr || typeof horaStr !== 'string') return null;
  
  // Limpiar el string y quitar segundos si los hay (HH:MM:SS -> HH:MM)
  const hora = horaStr.trim();
  
  // Diferentes patrones de hora
  const patrones = [
    /^(\d{1,2}):(\d{2})(:(\d{2}))?$/, // HH:MM o H:MM con segundos opcionales
    /^(\d{1,2})\.(\d{2})$/, // HH.MM o H.MM 
    /^(\d{1,2})h(\d{2})$/, // HHhMM o HhMM
    /^(\d{1,2}):(\d{2})\s*(am|pm)$/i // HH:MM am/pm
  ];
  
  for (const patron of patrones) {
    const match = hora.match(patron);
    if (match) {
      let horas = parseInt(match[1]);
      const minutos = parseInt(match[2]);
      
      // Ajustar horas para formato AM/PM
      if (match[3] && match[3].toLowerCase() === 'pm' && horas < 12) {
        horas += 12;
      } else if (match[3] && match[3].toLowerCase() === 'am' && horas === 12) {
        horas = 0;
      }
      
      // Validar rango de horas y minutos
      if (horas < 0 || horas > 23 || minutos < 0 || minutos > 59) return null;
      
      // Formatear con dos dígitos
      return `${horas.toString().padStart(2, '0')}:${minutos.toString().padStart(2, '0')}`;
    }
  }
  
  return null;
}
 
/**
 * Convierte un número de serie de Excel a una fecha en formato YYYY-MM-DD
 * @param {string|number} numeroSerie - Número de serie de Excel (días desde 1/1/1900)
 * @returns {string|null} - Fecha en formato YYYY-MM-DD o null si es inválida
 */
function convertirFechaExcel(numeroSerie) {
  try {
    // Asegurarse de que es un número
    const diasDesde1900 = parseInt(numeroSerie);
    if (isNaN(diasDesde1900)) return null;
    
    // Excel cuenta los días desde 1/1/1900, y tiene un error con el año 1900 siendo bisiesto
    // Para corregir, hay que restar 1 a los números mayores a 60
    let diasCorregidos = diasDesde1900;
    if (diasCorregidos > 60) {
      diasCorregidos -= 1;
    }
    
    // Crear una fecha base (1/1/1900) y añadir los días
    const fechaBase = new Date(1900, 0, 1);
    fechaBase.setDate(fechaBase.getDate() + diasCorregidos - 1); // -1 porque Excel cuenta desde 1, no 0
    
    if (isValid(fechaBase)) {
      const fechaFormateada = format(fechaBase, 'yyyy-MM-dd');
      console.log(`[Procesar Nuevo Dimensionamiento] Convertida fecha Excel ${numeroSerie} a ${fechaFormateada}`);
      return fechaFormateada;
    }
    
    return null;
  } catch (error) {
    console.warn(`[Procesar Nuevo Dimensionamiento] Error al convertir fecha Excel: ${numeroSerie}`, error);
    return null;
  }
}

        horario: horaInicio && horaFin ? `${horaInicio} a ${horaFin}` : '',

          }

      // Eliminar turnos existentes para este mes y año
      await Turno.deleteMany({ mes: periodoArchivo.mes, anio: periodoArchivo.anio });

        const turno = new Turno({

          horario: turnoData.horario,
          horaInicioReal: turnoData.horaInicioReal,
          horaFinReal: turnoData.horaFinReal,
          tipoDia: turnoData.tipoDia,
          mes: turnoData.mes,
          anio: turnoData.anio,
          motivo: turnoData.motivo,
          archivoId: archivoGuardado._id,
          asesor: {
            nombre: turnoData.nombre,
            id: turnoData.nombre.replace(/\s/g, '_').toLowerCase()
          },

    const asesoresUnicos = [...new Set(turnosProcesados.map(t => t.nombre))];
    for (const nombreAsesor of asesoresUnicos) {

      const asesorExistente = await Asesor.findOne({ nombre: nombreAsesor });

          const turnoConFecha = turnosProcesados.find(t =>

// Función auxiliar para calcular la hora de fin de un refrigerio
function calcularFinRefrigerio(horaInicio, duracionMinutos) {
  if (horaInicio === 'N/A') return 'N/A';
  
  // Convertir HH:MM a minutos desde medianoche
  const [horas, minutos] = horaInicio.split(':').map(Number);
  const minutosInicio = horas * 60 + minutos;
  
  // Sumar duración
  const minutosFin = minutosInicio + duracionMinutos;
  
  // Convertir de vuelta a formato HH:MM
  const horasFin = Math.floor(minutosFin / 60);
  const minutosFin2 = minutosFin % 60;
  
  return `${String(horasFin).padStart(2, '0')}:${String(minutosFin2).padStart(2, '0')}`;
}
 
// Función para normalizar nombres de campos
function normalizarCamposRegistro(row) {
  // Mapear posibles variaciones de nombres de columnas
  const mapeoColumnas = {
    // Variaciones para asesor
    'asesor': ['asesor', 'nombre', 'nombre_asesor', 'colaborador', 'empleado', 'agente', 'nombre del asesor'],
    // Variaciones para supervisor
    'supervisor': ['supervisor', 'nombre_supervisor', 'responsable', 'jefe', 'team_leader', 'líder'],
    // Variaciones para skill
    'skill': ['skill', 'habilidad', 'grupo', 'team', 'equipo', 'campaña', 'campaign'],
    // Variaciones para fecha
    'fecha': ['fecha', 'date', 'dia', 'día', 'jornada', 'fecha_turno'],
    // Variaciones para tipo_dia
    'tipo_dia': ['tipo_dia', 'tipo', 'tipo de día', 'tipo_de_dia', 'jornada', 'tipo_jornada'],
    // Variaciones para inicio
    'inicio': ['inicio', 'hora_inicio', 'ingreso', 'entrada', 'hora_entrada', 'start'],
    // Variaciones para fin
    'fin': ['fin', 'hora_fin', 'salida', 'hora_salida', 'end'],
    // Variaciones para motivo
    'motivo': ['motivo', 'razón', 'razon', 'tipo', 'causa', 'observación', 'observacion', 'reason']
  };
  
  // Verificar cada campo y normalizar al nombre estándar
  for (const [campoEstandar, variaciones] of Object.entries(mapeoColumnas)) {
    // Si el campo estándar ya existe, no hacer nada
    if (row[campoEstandar] !== undefined) continue;
    
    // Buscar variaciones en las claves del objeto
    for (const variacion of variaciones) {
      // Considerar versiones en minúsculas y sin espacios
      const variacionNormalizada = variacion.toLowerCase().replace(/\s+/g, '');
      
      // Verificar todas las claves del objeto para encontrar coincidencias
      for (const clave of Object.keys(row)) {
        const claveNormalizada = clave.toLowerCase().replace(/\s+/g, '');
        
        // Si encontramos una coincidencia, copiar el valor al campo estándar
        if (claveNormalizada === variacionNormalizada || claveNormalizada.includes(variacionNormalizada)) {
          row[campoEstandar] = row[clave];
          console.log(`[Procesar Nuevo Dimensionamiento] Campo '${clave}' normalizado a '${campoEstandar}'`);
          break; // Salir del bucle de claves
        }
      }
      
      // Si ya encontramos y asignamos una variación, salir del bucle
      if (row[campoEstandar] !== undefined) break;
    }
  }
  
  return row;
}
